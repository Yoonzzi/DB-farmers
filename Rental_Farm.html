Based on Farm Rating

파주 희망농장 5.00

SELECT [ f_name, avg(fr_score) as Rating ]
FROM [ farm_rating NATURAL JOIN total_farm
     NATURAL join registered_farm ]
WHERE [ rf_openterm IS NOT NULL
     AND reg_state = 'POSSIBLE']
GROUP BY [ total_farm.f_no ]
ORDER BY [ fr_score and reg_date ] DESC LIMIT 1
